    Constants: E_ION = 13.6 * 1.602176634e-19 J (hydrogen ionization energy), H_PLANCK = 2*pi*HBAR, M_E_KG = M_ELECTRON * 1e6 * 1.602176634e-19 / (2.998e8)**2.
    build_ionization_table(n_points=500) -> tuple[np.ndarray, np.ndarray]:
      Create temps = np.logspace(log10(1500), log10(10000), n_points).
      For each T, compute Saha RHS: A = (1/n_b) * (2*pi*M_E_KG*K_B*T/H_PLANCK**2)^1.5 * exp(-E_ION/(K_B*T)).
      For n_b (baryon number density): use n_b = OMEGA_B0 * RHO_CRIT * (T/T_CMB0)^3 / M_PROTON_KG where M_PROTON_KG = 1.67262192e-27 kg. Import OMEGA_B0, T_CMB0 from constants and RHO_CRIT from cosmology.
      Solve quadratic x^2 + Ax - A = 0: x = (-A + sqrt(A^2 + 4A)) / 2, clamped [0,1].
      Return (temps, x_values).
    get_ionization_fraction(temperature: float, table: tuple) -> float:
      If T > 10000: return 1.0. If T < 1500: return 0.0.
      Otherwise np.interp(temperature, table[0], table[1]).

    **structure.py**: Implement Jeans mass and Press-Schechter.
    Import G, K_B, SIGMA_8, N_S, AGE_UNIVERSE, T_CMB0, OMEGA_M0 from constants.py.
    Import RHO_CRIT from cosmology.
    M_PROTON_KG = 1.67262192e-27.

    compute_jeans_mass(temperature: float, matter_density: float, ionized: bool = True) -> float:
      gamma = 5/3. mu = 0.6 if ionized else 1.22.
      c_s = sqrt(gamma * K_B * temperature / (mu * M_PROTON_KG)).
      M_J = (math.pi/6) * c_s**3 / (G**1.5 * matter_density**0.5).
      Return M_J in kg.

    build_collapsed_fraction_table(n_points=200) -> tuple[np.ndarray, np.ndarray]:
      Use scipy.special.erfc.
      delta_c = 1.686.
      sigma_8 = SIGMA_8.
      n_s = N_S.
      cosmic_times = np.logspace(log10(6.3e15), log10(4.35e17), n_points) -- from First Stars to LSS.
      For each t, compute redshift z from scale factor a: a = (t/AGE_UNIVERSE)^(2/3) (matter-dominated approx), z = 1/a - 1.
      Growth factor D(z) ~ a = 1/(1+z).
      sigma_M = sigma_8 * D(z) (at M_8 scale, simplification for visualization).
      F_collapsed = erfc(delta_c / (sqrt(2) * sigma_M)).
      Return (cosmic_times, fractions).

    get_collapsed_fraction(cosmic_time: float, table: tuple) -> float:
      np.interp with clamp to table bounds.

    ```python
    @dataclass(frozen=True)
    class EraVisualConfig:
        """Visual configuration for one cosmological era.

        Drives per-era shader selection and uniform values.
        Lives in simulation layer (data only, no rendering imports).
        """
        era_index: int
        shader_key: str           # Key into ParticleSystem.programs dict
        base_color: tuple[float, float, float]
        accent_color: tuple[float, float, float]
        particle_size: float      # Point scale multiplier (1.0 = default 50.0)
        bloom_strength: float     # Post-processing bloom intensity (0.0-1.0)
        bloom_threshold: float    # Bright-pass threshold (higher = less bloom)
        expansion_rate: float     # Compute shader expansion modifier
        noise_strength: float     # Turbulence/noise contribution
        gravity_strength: float   # Gravitational clustering force
        damping: float            # Velocity damping factor
        brightness: float         # Fragment shader brightness multiplier
        transition_seconds: float # Duration of crossfade into this era
    ```

    **era_utils.glsl -- New shared era utility include:**
    ```glsl
    // Shared utility functions for per-era fragment shaders

    // Soft circular glow falloff for point sprites
    float soft_glow(vec2 point_coord, float softness) {
        float dist = length(point_coord - vec2(0.5));
        if (dist > 0.5) discard;
        return 1.0 - smoothstep(0.0, softness, dist);
    }

    // Sharp circular falloff for distinct particle edges
    float sharp_glow(vec2 point_coord, float inner, float outer) {
        float dist = length(point_coord - vec2(0.5));
        if (dist > 0.5) discard;
        return 1.0 - smoothstep(inner, outer, dist);
    }

    // Smooth easing for transition blend factors
    float smoothstep_ease(float t) {
        t = clamp(t, 0.0, 1.0);
        return t * t * (3.0 - 2.0 * t);
    }
    ```

    **particle_update.comp -- Upgrade to uniform-driven per-era behavior:**
    Replace the entire compute shader. Keep #version 430, layout, common.glsl include, SSBO declarations.
    ADD: `#include "noise.glsl"` after common.glsl include.
    New main():
    ```glsl
    void main() {
        uint idx = gl_GlobalInvocationID.x;
        if (idx >= u_particle_count) return;

        Particle p = particles_in[idx];
        vec3 pos = p.position.xyz;
        vec3 vel = p.velocity.xyz;
        float life = p.position.w;

        // Per-era expansion (Hubble-like flow)
        vel += pos * u_expansion_rate * u_dt;

        // Per-era turbulence via 3D simplex noise
        if (u_noise_strength > 0.001) {
            vec3 noise_sample = vec3(
                snoise3(pos * 0.5 + vec3(u_era_progress * 1.5, 0.0, 0.0)),
                snoise3(pos * 0.5 + vec3(0.0, u_era_progress * 1.5, 0.0)),
                snoise3(pos * 0.5 + vec3(0.0, 0.0, u_era_progress * 1.5))
            );
            vel += noise_sample * u_noise_strength * u_dt;
        }

        // Per-era gravitational clustering (late eras)
        if (u_gravity_strength > 0.001) {
            float r = max(length(pos), 0.1);
            vel += -normalize(pos) * u_gravity_strength * u_dt / r;
        }

        // Per-era damping
        vel *= (1.0 - u_damping * u_dt);

        // Integrate position
        pos += vel * u_dt;

        // Age particle
        life = clamp(life - 0.001 * u_dt, 0.0, 1.0);

        particles_out[idx].position = vec4(pos, life);
        particles_out[idx].velocity = vec4(vel, p.velocity.w);
        particles_out[idx].color = p.color;
    }
    ```

    **era_crossfade.frag -- New FBO crossfade composite shader:**
    ```glsl
    #version 430

    in vec2 v_texcoord;
    out vec4 fragColor;

    uniform sampler2D u_outgoing;    // Outgoing era's rendered scene
    uniform sampler2D u_incoming;    // Incoming era's rendered scene
    uniform float u_blend_factor;    // 0.0 = fully outgoing, 1.0 = fully incoming

    void main() {
        vec4 out_color = texture(u_outgoing, v_texcoord);
        vec4 in_color = texture(u_incoming, v_texcoord);
        fragColor = mix(out_color, in_color, u_blend_factor);
    }
    ```
    This shader uses the same fullscreen.vert as the other postprocess shaders (v_texcoord output).

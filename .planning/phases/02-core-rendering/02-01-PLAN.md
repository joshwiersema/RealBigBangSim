class BigBangSimApp(moderngl_window.WindowConfig):
    resource_dir = Path(__file__).parent / "shaders"

    def _load_shader(self, name: str) -> moderngl.Program:
        """Load vertex + fragment shader pair from resource_dir."""
        vert_path = self.resource_dir / f"{name}.vert"
        frag_path = self.resource_dir / f"{name}.frag"
        return self.ctx.program(
            vertex_shader=vert_path.read_text(),
            fragment_shader=frag_path.read_text(),
        )

       // Common uniforms for particle system

       uniform float u_scale_factor;
       uniform uint u_particle_count;
       uniform int u_era;
       uniform float u_era_progress;

       - `vec3 mod289(vec3 x)` helper
       - `vec3 permute(vec3 x)` helper

    3. `colormap.glsl` -- Temperature to RGB color mapping:
       - `vec3 temperature_to_color(float temp_kelvin)` -- Maps Kelvin temperature to approximate blackbody RGB color. Use a simplified Planckian locus approximation:
         - T < 3500K: red-orange (vec3(1.0, 0.4 + 0.3*(T/3500), 0.1))
         - 3500K-6500K: warm white (lerp from orange-white to pure white)
         - 6500K-10000K: cool white (lerp from white to blue-white)
         - T > 10000K: blue-white (vec3(0.6, 0.7, 1.0))
       - `vec3 density_to_color(float density_normalized)` -- Maps 0-1 density to blue-purple palette for cool/late eras

    **Compute shader** (`bigbangsim/shaders/compute/particle_update.comp`):
    ```glsl
    #version 430
    layout(local_size_x = 256) in;
    #include "common.glsl"

    layout(std430, binding = 0) readonly buffer ParticlesIn {
        Particle particles_in[];
    };
    layout(std430, binding = 1) writeonly buffer ParticlesOut {
        Particle particles_out[];
    };

    void main() {
        uint idx = gl_GlobalInvocationID.x;
        if (idx >= u_particle_count) return;

        Particle p = particles_in[idx];
        vec3 pos = p.position.xyz;
        vec3 vel = p.velocity.xyz;
        float life = p.position.w;

        // Hubble-like expansion flow: velocity proportional to distance
        float expansion_rate = u_scale_factor * 0.01;
        vel += pos * expansion_rate * u_dt;

        // Damping to prevent runaway
        vel *= (1.0 - 0.01 * u_dt);

        // Integrate position
        pos += vel * u_dt;

        // Age particle
        life = clamp(life - 0.001 * u_dt, 0.0, 1.0);

        particles_out[idx].position = vec4(pos, life);
        particles_out[idx].velocity = vec4(vel, p.velocity.w);
        particles_out[idx].color = p.color;  // Color updated by fragment shader, not compute
    }
    ```

    **Particle vertex shader** (`bigbangsim/shaders/vertex/particle.vert`):
    ```glsl
    #version 430

    uniform mat4 u_projection;
    uniform mat4 u_view;
    uniform float u_point_scale;  // Default 50.0

    struct Particle {
        vec4 position;
        vec4 velocity;
        vec4 color;
    };

    layout(std430, binding = 0) readonly buffer Particles {
        Particle particles[];
    };

    out vec4 v_color;
    out float v_life;

    void main() {
        Particle p = particles[gl_VertexID];
        gl_Position = u_projection * u_view * vec4(p.position.xyz, 1.0);
        gl_PointSize = u_point_scale / max(gl_Position.w, 0.01);
        v_color = p.color;
        v_life = p.position.w;
    }
    ```

    **Fragment shader variant 1** (`bigbangsim/shaders/fragment/particle_hot.frag`) -- early eras (Planck through Nucleosynthesis):
    ```glsl
    #version 430
    #include "colormap.glsl"

    in vec4 v_color;
    in float v_life;
    out vec4 fragColor;

    uniform float u_temperature;

    void main() {
        // Soft circular particle (distance from point center)
        vec2 coord = gl_PointCoord - vec2(0.5);
        float dist = length(coord);
        if (dist > 0.5) discard;

        float alpha = 1.0 - smoothstep(0.0, 0.5, dist);
        vec3 temp_color = temperature_to_color(u_temperature);

        // Blend vertex color with temperature-derived color
        vec3 final_color = mix(v_color.rgb, temp_color, 0.6);

        // Boost brightness for additive blending glow effect
        final_color *= 1.5 * v_life;

        fragColor = vec4(final_color, alpha * v_life);
    }
    ```

    **Fragment shader variant 2** (`bigbangsim/shaders/fragment/particle_cool.frag`) -- later eras (Dark Ages through LSS):
    ```glsl
    #version 430
    #include "colormap.glsl"

    in vec4 v_color;
    in float v_life;
    out vec4 fragColor;

    uniform float u_density_normalized;  // 0-1 normalized density

    void main() {
        vec2 coord = gl_PointCoord - vec2(0.5);
        float dist = length(coord);
        if (dist > 0.5) discard;

        float alpha = 1.0 - smoothstep(0.3, 0.5, dist);  // Sharper falloff than hot
        vec3 cool_color = density_to_color(u_density_normalized);

        vec3 final_color = mix(v_color.rgb, cool_color, 0.4);
        final_color *= v_life;

        fragColor = vec4(final_color, alpha * v_life * 0.8);
    }
    ```

    4. `fullscreen.vert` -- Passthrough for fullscreen quad (matches geometry.quad_fs() attributes):

    5. `bright_extract.frag` -- Bright-pass extraction for bloom:
       ```glsl
       #version 430
       in vec2 v_texcoord;
       out vec4 fragColor;
       uniform sampler2D u_scene;
       uniform float u_threshold;  // Default 1.0
       void main() {
           vec3 color = texture(u_scene, v_texcoord).rgb;
           float brightness = dot(color, vec3(0.2126, 0.7152, 0.0722));
           fragColor = (brightness > u_threshold) ? vec4(color, 1.0) : vec4(0.0, 0.0, 0.0, 1.0);
       }
       ```

    6. `gaussian_blur.frag` -- Separable Gaussian blur with hardcoded 5-tap weights:
       ```glsl
       #version 430
       in vec2 v_texcoord;
       out vec4 fragColor;
       uniform sampler2D u_image;
       uniform bool u_horizontal;
       const float weight[5] = float[](0.227027, 0.1945946, 0.1216216, 0.054054, 0.016216);
       void main() {
           vec2 tex_offset = 1.0 / textureSize(u_image, 0);
           vec3 result = texture(u_image, v_texcoord).rgb * weight[0];
           if (u_horizontal) {
               for (int i = 1; i < 5; ++i) {
                   result += texture(u_image, v_texcoord + vec2(tex_offset.x * i, 0.0)).rgb * weight[i];
                   result += texture(u_image, v_texcoord - vec2(tex_offset.x * i, 0.0)).rgb * weight[i];
               }
           } else {
               for (int i = 1; i < 5; ++i) {
                   result += texture(u_image, v_texcoord + vec2(0.0, tex_offset.y * i)).rgb * weight[i];
                   result += texture(u_image, v_texcoord - vec2(0.0, tex_offset.y * i)).rgb * weight[i];
               }
           }
           fragColor = vec4(result, 1.0);
       }
       ```

    7. `tonemap.frag` -- HDR tone mapping + bloom composite:
       ```glsl
       #version 430
       in vec2 v_texcoord;
       out vec4 fragColor;
       uniform sampler2D u_scene;
       uniform sampler2D u_bloom;
       uniform float u_exposure;        // Default 1.0
       uniform float u_bloom_strength;  // Default 0.3
       void main() {
           const float gamma = 2.2;
           vec3 hdrColor = texture(u_scene, v_texcoord).rgb;
           vec3 bloomColor = texture(u_bloom, v_texcoord).rgb;
           hdrColor += bloomColor * u_bloom_strength;
           vec3 mapped = vec3(1.0) - exp(-hdrColor * u_exposure);
           mapped = pow(mapped, vec3(1.0 / gamma));
           fragColor = vec4(mapped, 1.0);
       }
       ```
